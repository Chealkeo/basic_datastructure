## 思路
计算每一个海水可以连通的岛屿个数，取最大值。
## 复杂度
时间复杂度: O(N^4)
空间复杂度: O(N^2)
```python
class Solution:
    def largestIsland(self, grid: List[List[int]]) -> int:
        m = len(grid)
        if m==0: return 0
        n = len(grid[0])
        directions = [[0,1],[1,0],[0,-1],[-1,0]]
        res = 0
        
        def check(i,j):
            nonlocal res
            stk = [(i,j)]
            union = {(i,j)}
            while stk:
                i,j = stk.pop()
                for dx,dy in directions:
                    tmpx = i+dx
                    tmpy = j+dy
                    if 0<=tmpx<m and 0<=tmpy<n and (tmpx,tmpy) not in union and grid[tmpx][tmpy]==1:
                        stk.append((tmpx,tmpy))
                        union.add((tmpx,tmpy))
                        
            res = max(res,len(union))
        
        flag = False
        for i in range(m):
            for j in range(n):                
                if grid[i][j]==0:  
                    flag = True    
                    check(i,j)
        return res if flag else m*n
```
