## 题目
给定一个字符串 s，找到 s 中最长的回文子串。你可以假设 s 的最大长度为 1000。

示例 1：

输入: "babad"
输出: "bab"
注意: "aba" 也是一个有效答案。
示例 2：

输入: "cbbd"
输出: "bb"
## 思路
法一，中心扩展法
法二，动态规划法
## 代码
```python
class Solution:
    def longestPalindrome(self, s: str) -> str:
        #method1
        if len(s)==0:
            return ""
        res = s[0]

        def extend(i,j,s):
            while (i>=0 and j<len(s) and s[i]==s[j]):
                i -= 1
                j += 1
            return s[i+1:j]

        for i in range(len(s)-1):
            e1 = extend(i,i,s)#deal with odd nums
            e2 = extend(i,i+1,s)#deal with even nums
            print(e1,e2)
            e = max(len(e1),len(e2))
            if e > len(res):
                res = e1 if len(e1)>len(e2) else e2
        return res
        #method2
        #动态规划法，用dp[i][j]代表s[i:j]是否是回文串
        if not s: return ""
        n = len(s)
        ans = ""
        dp = [[False for _ in range(n)] for _ in range(n)]
        #l+1枚举子串可能的长度
        for l in range(n):
            #枚举子串可能的开始索引
            for i in range(n):
                j = i+l
                if j>=n:
                    break
                if l ==0:
                    dp[i][j]=True
                elif l==1:
                    dp[i][j]=(s[i]==s[j])
                else:
                    dp[i][j] = dp[i+1][j-1] and s[i]==s[j]
                if dp[i][j] and l+1>len(ans):
                    ans = s[i:j+1]
        return ans
```
